-- Utility function to completely clean up physical objects
local function clearPhysicalAccessories()
	local character = player.Character
	if character then
		for _, child in ipairs(character:GetChildren()) do
			if child:IsA("Accessory") and string.sub(child.Name, 1, 10) == "CustomLeg_" then
				child:Destroy()
			end
		end
	end
end

-- Render the physical objects onto the character model
local function applyAllAccessories()
	local character = player.Character
	if not character then return end
	
	local humanoid = character:WaitForChild("Humanoid", 5)
	if not humanoid or humanoid.RigType ~= Enum.HumanoidRigType.R6 then return end
	
	clearPhysicalAccessories()
	
	for index, data in ipairs(accessoryList) do
		local leg = character:FindFirstChild(data.targetLeg)
		if leg then
			local success, assetModel = pcall(function()
				return game:GetObjects("rbxassetid://" .. data.id)[1]
			end)
			
			if success and assetModel then
				local handle = assetModel:FindFirstChild("Handle") or assetModel:FindFirstChildWhichIsA("Part", true) or assetModel:FindFirstChildWhichIsA("MeshPart", true)
				if handle then
					handle.Anchored = false
					for _, obj in ipairs(assetModel:GetDescendants()) do
						if obj:IsA("BasePart") then
							obj.Anchored = false
							obj.CanCollide = false
							obj.Massless = true
						elseif obj:IsA("Weld") or obj:IsA("ManualWeld") or obj:IsA("WeldConstraint") or obj:IsA("Attachment") then
							obj:Destroy()
						elseif obj:IsA("SpecialMesh") then
							obj.Scale = Vector3.new(data.scaleX, data.scaleY, data.scaleZ)
						end
					end
					
					handle.Name = "Handle"
					handle.CanCollide = false
					handle.Massless = true
					
					if handle:IsA("MeshPart") or not handle:FindFirstChildWhichIsA("SpecialMesh") then
						handle.Size = Vector3.new(data.scaleX, data.scaleY, data.scaleZ)
					end
					
					local accessory = Instance.new("Accessory")
					accessory.Name = "CustomLeg_" .. tostring(index)
					handle.Parent = accessory
					
					local weld = Instance.new("Weld")
					weld.Name = "AccessoryWeld"
					weld.Part0 = handle
					weld.Part1 = leg
					
					local positionOffset = CFrame.new(data.offsetX, data.offsetY, data.offsetZ)
					local rotationOffset = CFrame.Angles(math.rad(data.rotateX), math.rad(data.rotateY), math.rad(data.rotateZ))
					
					weld.C0 = positionOffset * rotationOffset
					weld.Parent = handle
					
					accessory.Parent = character
				end
			end
		end
	end
end

-- Input Generator for Editor Grid (Retains Value fix)
local inputs = {}
local function createEditInput(labelName, property, x, y)
	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0, 75, 0, 22)
	label.Position = UDim2.new(0, x, 0, y)
	label.BackgroundTransparency = 1
	label.TextColor3 = Color3.fromRGB(200, 200, 200)
	label.Text = labelName
	label.TextSize = 13
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = editPanel

	local box = Instance.new("TextBox")
	box.Size = UDim2.new(0, 65, 0, 22)
	box.Position = UDim2.new(0, x + 80, 0, y)
	box.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
	box.TextColor3 = Color3.fromRGB(255, 255, 255)
	box.ClearTextOnFocus = false -- Minor fix requested: Stops automatic erasure
	box.TextSize = 13
	box.Parent = editPanel

	box.FocusLost:Connect(function()
		local data = accessoryList[selectedIndex]
		if data then
			data[property] = tonumber(box.Text) or data[property]
			box.Text = tostring(data[property])
		end
	end)
	inputs[property] = box
end

-- Interface Redraw Engine (Column 2)
updateEditUI = function()
	local data = accessoryList[selectedIndex]
	if not data then return end
	idBox.Text = tostring(data.id)
	for prop, box in pairs(inputs) do
		box.Text = tostring(data[prop])
	end
	if data.targetLeg == "Left Leg" then
		leftLegBtn.BackgroundColor3 = Color3.fromRGB(0, 130, 0)
		leftLegBtn.Text = "[Left Leg]"
		rightLegBtn.BackgroundColor3 = Color3.fromRGB(55, 55, 55)
		rightLegBtn.Text = "Right Leg"
	else
		leftLegBtn.BackgroundColor3 = Color3.fromRGB(55, 55, 55)
		leftLegBtn.Text = "Left Leg"
		rightLegBtn.BackgroundColor3 = Color3.fromRGB(0, 130, 0)
		rightLegBtn.Text = "[Right Leg]"
	end
end

-- List Construction Engine (Column 1 Manager)
updateSlotUI = function()
	slotPanel:ClearAllChildren()
	
	local layout = Instance.new("UIListLayout")
	layout.Padding = UDim.new(0, 4)
	layout.Parent = slotPanel
	
	for i, data in ipairs(accessoryList) do
		local sFrame = Instance.new("Frame")
		sFrame.Size = UDim2.new(1, 0, 0, 30)
		sFrame.BackgroundTransparency = 1
		sFrame.Parent = slotPanel
		
		local selBtn = Instance.new("TextButton")
		selBtn.Size = UDim2.new(0, 120, 1, 0)
		selBtn.BackgroundColor3 = (i == selectedIndex) and Color3.fromRGB(0, 90, 160) or Color3.fromRGB(40, 40, 40)
		selBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
		selBtn.Text = "Slot " .. tostring(i) .. " (" .. string.sub(data.targetLeg, 1, 1) .. ")"
		selBtn.Font = Enum.Font.SourceSans
		selBtn.TextSize = 13
		selBtn.Parent = sFrame
		
		selBtn.MouseButton1Click:Connect(function()
			selectedIndex = i
			updateSlotUI()
			updateEditUI()
		end)
		
		local delBtn = Instance.new("TextButton")
		delBtn.Size = UDim2.new(0, 45, 1, 0)
		delBtn.Position = UDim2.new(0, 125, 0, 0)
		delBtn.BackgroundColor3 = Color3.fromRGB(130, 30, 30)
		delBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
		delBtn.Text = "X"
		delBtn.Font = Enum.Font.SourceSansBold
		delBtn.Parent = sFrame
		
		delBtn.MouseButton1Click:Connect(function()
			if #accessoryList > 1 then
				table.remove(accessoryList, i)
				if selectedIndex > #accessoryList then selectedIndex = #accessoryList end
				updateSlotUI()
				updateEditUI()
			end
		end)
	end
	
	local addBtn = Instance.new("TextButton")
	addBtn.Size = UDim2.new(1, 0, 0, 30)
	addBtn.BackgroundColor3 = Color3.fromRGB(45, 85, 45)
	addBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
	addBtn.Text = "+ Add Accessory"
	addBtn.Font = Enum.Font.SourceSansBold
	addBtn.Parent = slotPanel
	addBtn.MouseButton1Click:Connect(function()
		table.insert(accessoryList, {id = 17121838966, targetLeg = "Left Leg", offsetX = 0, offsetY = 0, offsetZ = 0, rotateX = 0, rotateY = 0, rotateZ = 0, scaleX = 1, scaleY = 1, scaleZ = 1})
		selectedIndex = #accessoryList
		updateSlotUI()
		updateEditUI()
	end)
end

updatePresetUI = function()
	presetPanel:ClearAllChildren()
	local pLayout = Instance.new("UIListLayout")
	pLayout.Padding = UDim.new(0, 4)
	pLayout.Parent = presetPanel
	
	if not listfiles then return end
	local success, files = pcall(function() return listfiles(FOLDER_NAME) end)
	if not success or not files then return end
	
	for _, path in ipairs(files) do
		local filename = string.match(path, "([^/]+)%.txt$") or string.match(path, "([^\\]+)%.txt$")
		if filename then
			local pFrame = Instance.new("Frame")
			pFrame.Size = UDim2.new(1, 0, 0, 25)
			pFrame.BackgroundTransparency = 1
			pFrame.Parent = presetPanel
			
			local loadBtn = Instance.new("TextButton")
			loadBtn.Size = UDim2.new(1, 0, 1, 0)
			loadBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
			loadBtn.TextColor3 = Color3.fromRGB(230, 230, 230)
			loadBtn.Text = filename
			loadBtn.TextSize = 13
			loadBtn.Parent = pFrame
			
			loadBtn.MouseButton1Click:Connect(function()
				if readfile then
					local dataSuccess, content = pcall(function() return readfile(path) end)
					if dataSuccess then
						local tableSuccess, decoded = pcall(function() return HttpService:JSONDecode(content) end)
						if tableSuccess and type(decoded) == "table" then
							accessoryList = decoded
							selectedIndex = 1
							updateSlotUI()
							updateEditUI()
							applyAllAccessories()
						end
					end
				end
			end)
		end
	end
end
