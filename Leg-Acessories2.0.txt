local selectedIndex = 1

local function clearPhysicalAccessories()
	local character = player.Character
	if character then
		for _, child in ipairs(character:GetChildren()) do
			if child:IsA("Accessory") and (string.sub(child.Name, 1, 10) == "CustomLeg_" or child:GetAttribute("CustomLeg")) then
				pcall(function() child:Destroy() end)
			end
		end
	end
end

local function applyAllAccessories()
	local character = player.Character
	if not character then return end
	
	local humanoid = character:WaitForChild("Humanoid", 5)
	if not humanoid or humanoid.RigType ~= Enum.HumanoidRigType.R6 then return end
	
	clearPhysicalAccessories()
	
	for index, data in ipairs(accessoryList) do
		local targetLimbName = data.targetPart
		if not targetLimbName or (targetLimbName ~= "Left Leg" and targetLimbName ~= "Right Leg") then
			targetLimbName = "Left Leg"
			data.targetPart = "Left Leg"
		end

		local limb = character:FindFirstChild(targetLimbName)
		if limb then
			local success, assetModel = pcall(function()
				return game:GetObjects("rbxassetid://" .. data.id)[1]
			end)
			
			if success and assetModel then
				local wrapperModel = assetModel
				if not assetModel:IsA("Model") then
					wrapperModel = Instance.new("Model")
					assetModel.Parent = wrapperModel
				end

				local handle = wrapperModel:FindFirstChild("Handle") or wrapperModel:FindFirstChildWhichIsA("BasePart", true)
				if not handle then
					pcall(function() wrapperModel:Destroy() end)
					continue
				end
				handle.Name = "Handle"
				wrapperModel.PrimaryPart = handle

				for _, obj in ipairs(wrapperModel:GetDescendants()) do
					if obj:IsA("BasePart") then
						obj.Anchored = false
						obj.CanCollide = false
						obj.Massless = true
					elseif obj:IsA("Weld") or obj:IsA("ManualWeld") or obj:IsA("WeldConstraint") or obj:IsA("Attachment") then
						pcall(function() obj:Destroy() end)
					end
				end

				local targetScaleFactor = (data.scaleX + data.scaleY + data.scaleZ) / 3
				if targetScaleFactor <= 0 then targetScaleFactor = 1 end
				
				pcall(function()
					wrapperModel:Scale(targetScaleFactor)
				end)

				for _, obj in ipairs(wrapperModel:GetDescendants()) do
					if obj:IsA("BasePart") and obj ~= handle then
						local innerWeld = Instance.new("WeldConstraint")
						innerWeld.Name = "ModelInnerWeldConstraint"
						innerWeld.Part0 = obj
						innerWeld.Part1 = handle
						innerWeld.Parent = obj
					end
				end

				local accessory = Instance.new("Accessory")
				accessory.Name = "CustomLeg_" .. tostring(index)
				accessory:SetAttribute("CustomLeg", true)
				
				for _, child in ipairs(wrapperModel:GetChildren()) do
					child.Parent = accessory
				end
				pcall(function() wrapperModel:Destroy() end)

				local customHandle = accessory:FindFirstChild("Handle")
				if customHandle then
					local weld = Instance.new("Weld")
					weld.Name = "AccessoryWeld"
					weld.Part0 = customHandle
					weld.Part1 = limb
					
					local positionOffset = CFrame.new(data.offsetX, data.offsetY, data.offsetZ)
					local rotationOffset = CFrame.Angles(math.rad(data.rotateX), math.rad(data.rotateY), math.rad(data.rotateZ))
					
					weld.C0 = positionOffset * rotationOffset
					weld.Parent = customHandle
					
					accessory.Parent = character
				end
			end
		end
	end
end

local function updateDrag(input)
	local delta = input.Position - dragStart
	toggleBtn.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

toggleBtn.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = toggleBtn.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then dragging = false end
		end)
	end
end)

toggleBtn.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then updateDrag(input) end
end)

	if not data then return end
	
	idBox.Text = tostring(data.id)
	
	for prop, box in pairs(inputs) do
		box.Text = tostring(data[prop] or 0)
	end
	
	for limbName, btn in pairs(targetButtons) do
		if data.targetPart == limbName then
			btn.BackgroundColor3 = Color3.fromRGB(0, 130, 0)
			btn.Text = "[" .. limbName .. "]"
		else
			btn.BackgroundColor3 = Color3.fromRGB(55, 55, 55)

		local fileExists = false
		if readfile then
			pcall(function() fileExists = readfile(targetFilePath) ~= nil end)
		end
		
		local json = HttpService:JSONEncode(accessoryList)
		local success = false
		if writefile then
			success = pcall(function() writefile(targetFilePath, json) end)
		end
		
		if success then
			presetNameBox.Text = fileExists and "Updated!" or "Saved!"
			task.wait(1)
			presetNameBox.Text = "PresetName"
			updatePresetUI()
		else
			presetNameBox.Text = "No HW Save"
			task.wait(1)
			presetNameBox.Text = "PresetName"
		end
	else
		presetNameBox.Text = "Invalid Name"
		task.wait(1)
		presetNameBox.Text = "PresetName"
	end
end)

updatePresetUI = function()
	presetPanel:ClearAllChildren()
	
	local pLayout = Instance.new("UIListLayout")
	pLayout.Padding = UDim.new(0, 4)
	pLayout.Parent = presetPanel
	
	if not listfiles then return end
	
	local success, files
	success = pcall(function() files = listfiles(FOLDER_NAME) end)
	if not success or not files then return end
	
	for _, path in ipairs(files) do
		local filename = string.match(path, "([^/]+)%.txt$") or string.match(path, "([^\\]+)%.txt$")
		if filename then
			local pFrame = Instance.new("Frame")
			pFrame.Size = UDim2.new(1, 0, 0, 25)
			pFrame.BackgroundTransparency = 1
			pFrame.Parent = presetPanel
			
			local loadBtn = Instance.new("TextButton")
			loadBtn.Size = UDim2.new(1, 0, 1, 0)
			loadBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
			loadBtn.TextColor3 = Color3.fromRGB(230, 230, 230)
			loadBtn.Text = filename
			loadBtn.TextSize = 13
			loadBtn.Parent = pFrame
			
			loadBtn.MouseButton1Click:Connect(function()
				if not readfile then return end
				local dataSuccess, content = pcall(function() return readfile(path) end)
				if dataSuccess then
					local tableSuccess, decoded = pcall(function() return HttpService:JSONDecode(content) end)
					if tableSuccess and type(decoded) == "table" then
						
						-- Translation Engine mapping old structures and targetLeg naming rules safely
						for slotIdx, config in ipairs(decoded) do
							if type(config) == "table" then
								-- Handle legacy partsData formats
								if config.partsData then
									local hData = config.partsData["Handle"]
									if hData then
										config.offsetX = hData.offsetX or config.offsetX or 0
										config.offsetY = hData.offsetY or config.offsetY or 0
										config.offsetZ = hData.offsetZ or config.offsetZ or 0
										config.rotateX = hData.rotateX or config.rotateX or 0
										config.rotateY = hData.rotateY or config.rotateY or 0
										config.rotateZ = hData.rotateZ or config.rotateZ or 0
										config.scaleX = hData.scaleX or config.scaleX or 1
										config.scaleY = hData.scaleY or config.scaleY or 1
										config.scaleZ = hData.scaleZ or config.scaleZ or 1
									end
									config.partsData = nil
								end

								-- Map historical targetLeg variables directly over to modern targetPart strings
								if config.targetLeg and not config.targetPart then
									config.targetPart = config.targetLeg
								end

								if not config.targetPart or (config.targetPart ~= "Left Leg" and config.targetPart ~= "Right Leg") then
									config.targetPart = "Left Leg"
								end
							end
						end
						
						accessoryList = decoded
						selectedIndex = 1
						updateSlotUI()
						updateEditUI()
						applyAllAccessories()
					end
				end
			end)
		end
	end
end
